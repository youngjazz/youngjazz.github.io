Leon' Blog
2019-03-16
docker

RabbitMq

docker run --name rabbitmq -d --hostname my-rabbit -p 5672:5672 -p 15672:15672 rabbitmq:3.7.8-management

Redis

docker run -d -p 6379:6379 redis:4.0.8

Zipkin

docker run -d -p 9411:9411 openzipkin/zipkin

MySql MYSQL_ROOT_HOST=127.17.%.%

docker run --name mysql2 --env MYSQL_ROOT_HOST=% --env MYSQL_ROOT_PASSWORD=123456 -v /Users/leon/data/mysql_data:/var/lib/mysql -p 3306:3306 -d mysql/mysql-server:5.7

activeMQ —rm指容器停止后会被删除

docker run --name='activemq' -d --rm -p 61616:61616 -p 8161:8161 webcenter/activemq:latest

maven私服

docker run -d -p 8081:8081 --name nexus -v /Users/leon/data/nexus-data:/var/nexus-data sonatype/nexus3
